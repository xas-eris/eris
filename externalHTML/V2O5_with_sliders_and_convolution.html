import numpy as np
from bokeh.io import output_notebook, show

output_notebook(hide_banner=True)

#import data as a text file:
dataFile = open('space_dust_V2O5.txt','r')

#skip header information:
for aline in dataFile:
    if aline == '#  energy  counttime  i0  itrans\n':
        break

#put all the data in a list and close the file:
dataList = dataFile.readlines()
dataFile.close()

#create empty lists:
energyList = []
modifiedEnergyList = [] #this list will be the same as energyList but horizontally offset
muList = []
i0List = []
iTList = []

#fill lists with appropriate values:
for dataPoint in dataList:
    energyValue = float(dataPoint[2:15])
    energyList.append(energyValue)
    modifiedEnergyList.append(energyValue - 5465)
    
    i0Value = float(dataPoint[36:49])
    i0List.append(i0Value)
    itransValue = float(dataPoint[53:66]) #transmitted intensity if there are no holes
    iTList.append(itransValue)
    muValue = - np.log(itransValue/i0Value) #The TRUE (mu * thickness) value
    muList.append(muValue)

#truncate the last data point from the lists if the lists are of even length; this prevents the convolution from being horizontally offset by one point.
if len(energyList)%2 == 0:
    energyList = energyList[:-1]
    muList = muList[:-1]
    i0List = i0List[:-1]
    iTList = iTList[:-1]

#subtract linear offset:
b = muList[0]
for ii in range(len(muList)):
    muList[ii] = muList[ii] - b

#Choose deltaE, the horizontal separation between points in the to-be interpolated data, to be the smallest existing separation between points in the non-interpolated data:
deltaE = ""
for i in range(1,len(energyList)):
    if deltaE == "" or abs(energyList[i] - energyList[i-1]) < deltaE:
        deltaE = energyList[i] - energyList[i-1]

#The number of points in the interpolated data:
arrayLength = int( np.floor( (energyList[len(energyList) - 1] - energyList[0]) / deltaE ) )

#Create a list of the interpolated energy (i.e., x) values:
interpolatedX = []
firstXValue = np.ceil(energyList[0])
for i in range(arrayLength):
    interpolatedX.append(firstXValue + i * deltaE)

#Create a list that is just muList but normalized:
normalizedMuList = []
ii = 0
while energyList[ii] < 5500: #finds the index corresponding to an energy value of normEnergy (eV)
        ii += 1
diff = muList[ii]-muList[0]
for ii in range(len(muList)):
    normalizedMuList.append(muList[ii] / diff)

source = ColumnDataSource(data={'x':energyList,'y':muList, 'normalizedy':normalizedMuList}) # y will change as sliders are manipulated; mu (and all other of these dictionary entries) will never change.

slider = Slider(start=0, end=1, value=0, step=.01, title='alpha')
checkbox1 = CheckboxGroup(labels=["Normalize"])
slider2 = Slider(start=0.1, end=5,value=1,step=0.1,title='thickness factor')
checkbox2 = CheckboxGroup(labels=["Convolve with Lorentzian"])
#slider3 = Slider(start=0.01, end=1,value=0.01,step=0.02,title='Lorentzian FWHM')
slider3 = Slider(start=0.1, end=15,value=0.1,step=0.1,title='Lorentzian FWHM')
alpha_txt = TextInput(value="0", title="",css_classes=['customTextInput'])
th_txt = TextInput(value="1", title="",css_classes=['customTextInput'])
fwhm_txt = TextInput(value="0.1",title="",css_classes=['customTextInput'])

txtVals = ColumnDataSource(data={'txtdata':[alpha_txt.value, th_txt.value, fwhm_txt.value]})

update_curve = CustomJS(args=dict(source=source, source2=source2, slider=slider, checkbox1=checkbox1, slider2=slider2, checkbox2=checkbox2, slider3=slider3, alpha_txt=alpha_txt, th_txt=th_txt, fwhm_txt=fwhm_txt, txtVals=txtVals), code="""
    var changedata = txtVals.data;
    txtdata = changedata['txtdata'];
    
    // Set alpha depending on whether its slider or text input was adjusted:
    // First, if the slider was adjusted...
    if (Number(txtdata[0]) === Number(alpha_txt.value)) {
        var alpha = slider.value;
        alpha_txt.value = '' + alpha;
        txtdata[0] = alpha;
    }
    // Else, the text input was changed...
    else {
        var alpha = Number(alpha_txt.value);
        slider.value = alpha;
        txtdata[0] = alpha;
    }
    
    // Set thicknessFactor depending on whether its slider or text input was adjusted:
    if (Number(txtdata[1]) === Number(th_txt.value)) {
        var thicknessFactor = slider2.value;
        th_txt.value = '' + thicknessFactor;
        txtdata[1] = thicknessFactor;
    }
    else {
        var thicknessFactor = Number(th_txt.value);
        slider2.value = thicknessFactor;
        txtdata[1] = thicknessFactor;
    }
    
    // Set FWHM depending on whether its slider or text input was adjusted:
    if (Number(txtdata[2]) === Number(fwhm_txt.value)) {
        var FWHM = slider3.value;
        fwhm_txt.value = '' + FWHM;
        txtdata[2] = FWHM;
    }
    else {
        var FWHM = Number(fwhm_txt.value);
        slider3.value = FWHM;
        txtdata[2] = FWHM;
    }
    
    var normalize = JSON.stringify(checkbox1.active);
    normalize = Number(normalize[1]); // makes normalize equal to 0 if the checkbox is checked; otherwise its value is NaN.
    var useLorentzian = JSON.stringify(checkbox2.active);
    useLorentzian = Number(useLorentzian[1]);
    var data = source.data;
    var data2 = source2.data;
    x = data['x'];
    y = data['y'];

    mu = data2['mu'];
    i0 = data2['i0'];
    iT = data2['iT'];
    
    // Convolve data if checkbox is checked:
    if (useLorentzian === 0) {
        // Choose deltaE, the horizontal separation between points in the to-be interpolated data, to be the smallest existing separation between points in the non-interpolated data:
        var deltaE = "";
        for (var i = 1; i < x.length; i++) {
            if (Math.abs(x[i] - x[i-1]) < deltaE || deltaE === ""){
                deltaE = x[i] - x[i-1];
            }
        }
        
        // The number of points in the interpolated data:
        arrayLength = Math.floor( (x[x.length - 1] - x[0]) / deltaE );
        
        // Create a list of the interpolated energy (i.e., x) values:
        var interpolatedX = [];
        var firstXValue = Math.ceil(x[0]);
        for (var i = 0; i < arrayLength; i++) {
            interpolatedX.push(firstXValue + i * deltaE);
        }
        
        // Create a list of thicknessFactor-dependent, and alpha-dependent, simulated IT data:
        var simulatedIT = [];
        for (var i = 0; i < x.length; i++) {
            simulatedIT.push( alpha * i0[i] + (1-alpha) * i0[i] * Math.pow(Math.E, - thicknessFactor * mu[i] ) );
        }
        
        // Interpolate simulatedIT:
        var interpolatedIT = [];
        j = 0;
        for (var i = 1; i < x.length; i++) {
            iTInitial = simulatedIT[i-1];
            iTFinal = simulatedIT[i];
            xInitial = x[i-1];
            xFinal = x[i];
            
            var interpolatingSlope = (iTFinal - iTInitial) / (xFinal - xInitial);
            while (interpolatedX[j] <= xFinal) {
                interpolatedIT.push( iTInitial + interpolatingSlope * (interpolatedX[j] - xInitial) );
                j++;
            }
        }
        
        // Interpolate i0:
        var interpolatedI0 = [];
        j = 0;
        for (var i = 1; i < x.length; i++) {
            i0Initial = i0[i-1];
            i0Final = i0[i];
            xInitial = x[i-1];
            xFinal = x[i];
            
            var interpolatingSlope = (i0Final - i0Initial) / (xFinal - xInitial);
            while (interpolatedX[j] <= xFinal) {
                interpolatedI0.push( i0Initial + interpolatingSlope * (interpolatedX[j] - xInitial) );
                j++;
            }
        }
        
        // Create a list of Lorentzian values:
        function lorentzian(x,x0) {
            return FWHM / (2 * Math.PI * ((x - x0)*(x - x0) + FWHM*FWHM/4));
        }
        
        // Convolve the iT and i0 data with the Lorentzian:
        for (var i = 0; i < x.length; i++) {
            var iTaccumulator = 0;
            var i0accumulator = 0;
            var lorCenter = x[i];
            for (var j = 0; j < arrayLength; j++) {
                lorValue = lorentzian(interpolatedX[j],lorCenter);
                iTaccumulator += interpolatedIT[j] * lorValue; // accumulator += interpolatedIT[j] * lorList[arrayLength - 1 - i + j]; could work if we end up displaying interpolated data instead of only using interpolated data for the integration (as we're presently doing). In that case, the iteration variable "i" would need to go all the way to arrayLength instead of merely x.length
                i0accumulator += interpolatedI0[j] * lorValue;
            }
            // calculate mu:
            y[i] = - Math.log(iTaccumulator/i0accumulator);
        }
        
    }
    
    // when the convolution checkbox gets un-checked, re-calculate mu and apply pinhole effect:
    else {
        for (var i = 0; i < x.length; i++) {
            var modifiedItransValue = alpha + (1-alpha) * Math.pow(Math.E, - thicknessFactor * mu[i]);
            y[i] = - Math.log(modifiedItransValue);
        }
    }
    
    // normalize the data:
    if (normalize === 0) {
        // first determine the index corresponding to the first energy value greater than 5500:
        var index = -1;
        x.some(function(el, i) {
            if (el > 5500) {
                index = i;
                return true;
            }
        });
        
        var diff = y[index] - y[0];
        for (var i = 0; i < x.length; i++) {
            if (alpha != 1) {
                y[i] /= diff;
            }
        }
    }
    
    // subtract vertical offset:
    var b = y[0];
    for (var i = 0; i < x.length; i++) {
        y[i] -= b;
    }
    
    // necessary becasue we mutated source.data in-place:
    source.change.emit();
    txtVals.change.emit();
    slider.change.emit();
    slider2.change.emit();
""")
